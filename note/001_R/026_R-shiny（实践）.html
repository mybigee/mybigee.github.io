# 定义UI
ui <- fluidPage(
  sidebarLayout(
    # 输入：选择变量绘制
    sidebarPanel(
      # 选择变量对y轴
      selectInput(
        inputId = "y",
        label = "Y-axis:",
        choices = c("imdb_rating", "imdb_num_votes", "critics_score", "audience_score", "runtime"),
        selected = "audience_score" # 设置默认选中的选项
      ),
      # 选择变量对x轴
      selectInput(
        inputId = "x",
        label = "X-axis:",
        choices = c("imdb_rating", "imdb_num_votes", "critics_score", "audience_score", "runtime"),
        selected = "critics_score"
      )
    ),
    # 输出：显示散点图
    mainPanel(
      plotOutput(outputId = "scatterplot")
    )
  )
)
# 定义服务端
server <- function(input, output, session) {
  output$scatterplot <- renderPlot({
    ggplot(data = movies, aes_string(x = input$x, y = input$y)) +
      geom_point()
  })
}
# 创建一个shiny app对象
shinyApp(ui = ui, server = server)

2.2d Convert dateInput() to dateRangeInput()

  • 使用dateRangeInput()，可以得到一个长度为2的向量，第一个元素是开始日期，第二个元素是结束日期。
  • 查看dateInput和dateRangeInput的帮助文档，将dateInput改为dateRangeInput，指定开始和结束日期为2013 年 1 月 1 日和 2014 年 1 月 1 日。
  • 将startview修改为year，方便用户阅读。
  • 添加必要的req()

}

server <- function(input, output, session) {

    ggplot(data = movies, aes_string(x = input$x, y = input$y)) +

  })

3-4c Trigger with observeEvent()

  • 点击按钮：一条消息打印到控制台，说明显示了多少条记录以及这些记录的表输出。
  • 填写必要的功能和输入 ID，设置一个表输出，该输出仅在单击操作按钮时打印，但在创建该输出的其他输入发生更改时不会打印。
x
# Load packages ----------------------------------------------------------------


library(shiny)


# Load data --------------------------------------------------------------------

# Define UI --------------------------------------------------------------------


ui <- fluidPage(
  sidebarLayout(
    sidebarPanel(
      numericInput(
        inputId = "n_rows",
        label = "How many rows do you want to see?",
        value = 10
      ),
      
      actionButton(
        inputId = "button",
        label = "Show"
      )
    ),
    
    mainPanel(
      tableOutput(outputId = "datatable")
    )
  )
)


# Define server ----------------------------------------------------------------


server <- function(input, output, session) {
  
  # Print a message to the console every time button is pressed;
  observeEvent(input$button, {
    cat("Showing", input$n_rows, "rows\n")
  })
  
  # Take a reactive dependency on input$button, but not on any other inputs
  df <- eventReactive(input$button, {
    head(movies, input$n_rows)
  })
  output$datatable  <- renderTable({
    df()
  })
  
}


# Create the Shiny app object --------------------------------------------------


shinyApp(ui = ui, server = server)
