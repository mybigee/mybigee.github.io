# DESeq2 差异分析
library(DESeq2)
# 读取reads count 数据
# row.names = 1 以第一列作为行名
countsTable <- read.delim("DE_data.tsv",header = F,row.names = 1)
# 添加列名
sampleNames <- paste0("sample",1:12)
colnames(countsTable) <- sampleNames
# 创建一个分组信息对象（数据框）
countsDesign <- data.frame(row.names = colnames(countsTable),
     group = c("FY","FG","FY","FG","FG","FY","MY","MY","MY","MG","MG","MG"))
head(countsDesign)
# 将reads count 数据由数据框转为矩阵类型
countsMatrix <- as.matrix(countsTable)
# 将矩阵转为DESeq2的数据类型dds
# 使用group来进行比较
dds <- DESeqDataSetFromMatrix(countsMatrix,colData = countsDesign,design = ~group)
# 除去所有值都为0的行
dds <- dds[rowSums(counts(dds))>0,]
# DESeq2 分析：大小因子估计、离差的估计、负二项分布的拟合以及计算相应的统计量
# parallel = T：进行并行计算，提高运行速度
dds <- DESeq(dds,parallel = T)
# 提取性别间差异表达基因的分析结果
sex_Y_res <- results(dds,contrast = c("group","FY","MY"),parallel = T)
sex_G_res <- results(dds,contrast = c("group","FG","MG"),parallel = T)
# 在群体间差异表达基因分析的结果
population_F_res <- results(dds,contrast = c("group","FY","FG"),parallel = T)
population_M_res <- results(dds,contrast = c("group","MY","MG"),parallel = T)
# 将性别间显著差表达的基因
Sig_sex_Y_0.01 <- sex_Y_res[which(sex_Y_res$padj < 0.01),]
write.csv(Sig_sex_Y_0.01,file = "Sig_sex_Y_0.01_DESeq2.csv")
# 提取种群间差异表达的基因
Sig_pop_F_0.01 <- population_F_res[which(population_F_res$padj < 0.01),]
write.csv(Sig_pop_F_0.01,file = "Sig_pop_F_0.01_DESeq2.csv")

# edgeR 差异表达分析
library(limma)
library(edgeR)
# 读取reads count 数据
# row.names = 1 以第一列作为行名
rawdata <- read.delim("DE_data.tsv",header = F,row.names = 1)
# 添加列名
sampleNames <- paste0("sample",1:12)
colnames(rawdata) <- sampleNames
# 创建一个分组信息对象（数据框）
countsDesign <- data.frame(row.names = colnames(rawdata),
    group = c("FY","FG","FY","FG","FG","FY","MY","MY","MY","MG","MG","MG"))
# 构建DGEList对象（edgeR所需的数据格式）
data <- DGEList(counts = rawdata,group = countsDesign$group)
# 过滤，至少有三个样本大于1
# cpm(data)：每一个reads数除以reads总数*1百万
keep <- rowSums(cpm(data)>1)>=3
# keep.lib.sizes 参数来控制归一化过程中是否保留样本大小因子
data <- data[keep,,keep.lib.sizes = F]
# 数据的标准化
data <- calcNormFactors(data)
# 创建设计矩阵
design <- model.matrix(~0+group,data = countsDesign)
head(design)
# 设计矩阵的列名
colnames(design) <- levels(factor(countsDesign$group))
# 估计方差
data <- estimateDisp(data,design)
# 构建DGEGLM对象
fit <- glmQLFit(data,design)
# 创建对照矩阵
my.contrasts <- makeContrasts(YRI.F.vs.M = FY - MY,GBR.F.vs.M = FG - MG,
                              F.YRI.vs.GBR = FY - FG,M.YRI.vs.GBR = MY - MG,levels = design)
my.contrasts
# 差异分析 QL F-test
qlf_YRI.F.vs.M <- glmQLFTest(fit,contrast = my.contrasts[,"YRI.F.vs.M"])
qlf_GBR.F.vs.M <- glmQLFTest(fit,contrast = my.contrasts[,"GBR.F.vs.M"])
qlf_F.YRI.vs.GBR <- glmQLFTest(fit,contrast = my.contrasts[,"F.YRI.vs.GBR"])
qlf_M.YRI.vs.GBR <- glmQLFTest(fit,contrast = my.contrasts[,"M.YRI.vs.GBR"])
# 对每个差异分析的p值进行校正（以qlf_YRI.F.vs.M为例）
YRI.F.vs.M_padj <- p.adjust(qlf_YRI.F.vs.M$table$PValue,method = "BH")
# 提取差异表达基因
Sig_YRI.F.vs.M <- qlf_YRI.F.vs.M$table[which(YRI.F.vs.M_padj < 0.05),]
# 保存到文件
write.csv(Sig_YRI.F.vs.M,"Sig_YRI.F.vs.M_0.05_edgeR.csv")

火山图

​x
# 差异分析
library(limma)
design <- model.matrix(~factor(group))
fit <- lmFit(experSet,design)
fit <- eBayes(fit)
# coef=2则表示获取第二个自变量的系数
DEG <- topTable(fit,coef = 2,n = Inf)
with(DEG, plot(logFC, -log10(P.Value), pch=20, main="Volcano plot"))  


# 差异分析的阈值：mean+2*sd
logFC_cutoff <- with(DEG,mean(abs(logFC)) + 2*sd(abs(logFC)))
DEG$change <- ifelse(DEG$P.Value < 0.05 & abs(DEG$logFC) > logFC_cutoff,
                     ifelse(DEG$logFC > logFC_cutoff ,'UP','DOWN'),"NOT")
this_title <- paste0('logFC的阈值为 ',round(logFC_cutoff,3),
                    '\n上调基因数量 ',nrow(DEG[DEG$change =='UP',]) ,
                    '\n下调基因数量 ',nrow(DEG[DEG$change =='DOWN',]))
ggplot(data = DEG,aes(x = logFC, y = -log10(P.Value), col = change)) +
  geom_point(alpha = 0.4, size = 1.75) +
  theme_set(theme_bw(base_size = 20)) +
  xlab("log2 fold change") + ylab("-log10 p-value") +
  ggtitle(this_title) + theme(plot.title = element_text(size = 15,hjust = 0.5)) +
  scale_colour_manual(values = c('blue','black','red'))
