12.差异分析（DEG）和火山图volcano plot

xxxxxxxxxx
# 差异分析
library(limma)
# 设计矩阵
design <- model.matrix(~factor(group))
head(design,3)
# 构建线性模型并对其进行拟合
fit <- lmFit(experSet,design)
# 进行贝叶斯检验
fit <- eBayes(fit)
# coef=2则表示获取第二个自变量的系数
DEG <- topTable(fit,coef = 2,n = Inf)
head(DEG,3)
with(DEG, plot(logFC, -log10(P.Value), pch=20, main="Volcano plot"))

# 差异分析的阈值：mean+2*sd
logFC_cutoff <- with(DEG,mean(abs(logFC)) + 2*sd(abs(logFC)))
DEG$change <- ifelse(DEG$P.Value < 0.05 & abs(DEG$logFC) > logFC_cutoff,
                     ifelse(DEG$logFC > logFC_cutoff ,'UP','DOWN'),"NOT")
this_title <- paste0('logFC的阈值为 ',round(logFC_cutoff,3),
                    '\n上调基因数量 ',nrow(DEG[DEG$change =='UP',]) ,
                    '\n下调基因数量 ',nrow(DEG[DEG$change =='DOWN',]))
ggplot(data = DEG,aes(x = logFC, y = -log10(P.Value), col = change)) +
  geom_point(alpha = 0.4, size = 1.75) +
  theme_set(theme_bw(base_size = 20)) +
  xlab("log2 fold change") + ylab("-log10 p-value") +
  ggtitle(this_title) + theme(plot.title = element_text(size = 15,hjust = 0.5)) +
  scale_colour_manual(values = c('blue','black','red'))
