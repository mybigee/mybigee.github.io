java -jar trimmomatic-0.32.jar PE -phread64 reads1.fastq.gz reads2.fastq.gz paired1.fq.gz unpaired1.fq.gz paired2.fq.gz unpaired2.fq.gz AVGQUAL:20

# reads 的生物信息学分析
# ① 创建索引：加快比对速度，bowtie2
# -f：指定参考基因组的个数为fasta（默认）
# charX.fa：参考基因组
# chrX：要创建的索引文件名（前缀）
bowtie2-build -f charX.fa chrX
# 创建的索引文件的扩展名为.bt2


# ② 与参考基因组比对：tophat2
# -p 8：指定8个线程吗，提高运行速率
# -G chrX.gtf：参考基因组注释文件
# -o ERR188044：指定输出的目录（不存在会自动创建）
# chrX：上一步创建的索引文件
# ERR188044_chrX_1/2.fastq.gz：要比对的双端reads文件，单端就一个文件即可
tophat2 -p 8 -G chrX.gtf -o ERR188044 chrX ERR188044_chrX_1.fastq.gz ERR188044_chrX_2.fastq.gz


# ③ 转录本的组装和定量：cufflinks
# -p 8：线程数
# -g 与 -G的区别是，-G只会组装并定量与参考基因组注释文件匹配的转录本，而-g还可以组装新的转录本
# accepted_hits.bam：输入文件，即上一步比对的输出文件
cufflinks -p 8 -g chrX.gtf -o ERR188044 accepted_hits.bam


# ④ 转录本合并：cuffmerge
# -s chrX.fa：参考基因组文件
# assemblies.txt：手动配置，内容为需要合并的转录本文件
cuffmerge -p 8 -g chrX.gtf -s chrX.fa assemblies.txt


# ⑤ 差异分析：cuffdiff（只能进行单因素分析）
# 因此这里只分析性别
# -L 进行成差异分析的两组样本的分析名（自己定）
# -u 上一步得到的合并之后的转录本
# 接下来的文件为参考基因组比对之后的bam文件，分为两组，使用逗号隔开
cuffdiff -p 8 -b chrX.fa -o 4_cuffdiff -L Female,Male -u merged_asm/merged.gtf ERR188245/accepted_hits.bam, ERR188428/accepted_hits.bam, ERR188337/accepted_hits.bam, ERR204916/accepted_hits.bam,ERR188234/accepted_hits.bam, ERR188273/accepted_hits.bam # 第一组（Female）样本
ERR188401/accepted_hits.bam, ERR188257/accepted_hits.bam, ERR188383/accepted_hits.bam, ERR188454/accepted_hits.bam, ERR188104/accepted_hits.bam,ERR188044/accepted_hits.bam # 第二组（Male）样本

# ① 创建索引：hisat2-build
# 已知的可变剪切信息
extract_splite_sites.py chrX.gtf > chrX.ss
# 已知的外显子信息
extract_exons.py chrX.gtf > chrX.exon
# []：可省略
# chrX.fa：参考基因组
# chrX：索引文件前缀
hisat2-build [--ss chrX.ss --exon chrX.exon] chrX.fa chrX
# 生成的文件的扩展名为.ht2


# ② 与参考基因组比对：hisat2
# --dta：为下一步的stringtie的组装准备，提高运行速度
# chrX：上一步索引文件的前缀
# -1 -2：表示双端的reads文件，如果是单端，使用-U 
# -S：指定输出的文件（sam格式）
hisat2 -p 8 --dta -x chrX -1 ERR188044_chrX_1.fastq.gz -2 ERR188044_chrX_2.fastq.gz -S ERR188044_chrX.sam


# ③ 对sam文件的处理：下一步的stringtie处理的文件为bam格式的文件，samtools
# sort：对转化后的bam文件进行排序，默认按reads再染色体上的位置
# -@ 8：指定线程数
# -o 输出文件
samtools sort -@ 8 -o ERR188044_chrX.bam ERR188044_chrX.sam


# ④ 转录本的组装：stringtie
# -o 组装后的转录本文件
# -l 输出的转录本名字的前缀（默认为STRD）
stringtie -p 8 -G chrX.gtf  -o ERR188044_chrX.gtf -l ERR188044 ERR188044_chrX.bam


# ⑤ 转录本的合并：stringtie --merge
# -o 合并后的输出文件
# mergelist.txt自己配置
stringtie --merge -p 8 -G chrX.gtf -o stringtie_merge.gtf mergelist.txt


# ⑥ 转录组的定量分析：stringtie
# -B：指定ballgown所需的文件
stringtie -B -p 8 -G string_merge.gtf -o ERR188044/ERR188044_chrX.gtf ERR188044_chrX.bam

# ballgown 差异分析
library(ballgown)
library(genefilter) # 计算均值和方差
library(dplyr)
# 读取样本的表型数据
pheno_data <- read.csv("geuvadis_phenodata.csv")
# 读取定量表达的数据（上一步定量分析的到的数据）
# samplePattern：匹配带有指定字符的文件
# pData：指定表型数据
bg_chrX <- ballgown(dataDir = "../4_ballgown/",
                    samplePattern = "ERR",pData = pheno_data)
# 过滤表达量低的基因
# rowVars(bg_chrX)>1：获取方差大于1的行
# genomesubset：根据基因进行筛选，而不是根据样本名
bg_chrX_filt <- subset(bg_chrX,"rowVars(bg_chrX)>1",genomesubset = T)
# 对转录本进行差异分析
results_transcripts <- stattest(bg_chrX_filt,feature = "transcript",
   covariate = "sex",adjustvars = c("population"),getFC = T,meas = "FPKM")
# 对基因进行差异分析
# covariate = "sex"：指定协变量，即剔除sex的影响
# adjustvars = c("population")：对不同的人群进行差异分析
# getFC：是否计算差异倍数
# meas = "FPKM"：衡量表达量的标准
results_transcripts <- stattest(bg_chrX_filt,feature = "transcript",
   covariate = "sex",adjustvars = c("population"),getFC = T,meas = "FPKM")
# 为转录本添加基因名字、ID
results_genes <- data.frame(geneNames = ballgown::geneNames(bg_chrX_filt),
                          geneIDs = ballgown(bg_chrX_filt),results_transcripts)
# 按结果的pvalue进行排序(升序排列)
results_transcripts <- arrange(results_transcripts,pval)
results_genes <- arrange(results_genes,pval)
# 提取显著差异表达的基因和转录本（q值小于0.05）
results_transcripts_0.05 <- subset(results_transcripts,results_transcripts$qval < 0.05)
results_genes_0.05 <- subset(results_genes,results_genes$qval < 0.05)
# 将显著差异表达的基因和转录本保存为csv文件
write.csv(results_transcripts_0.05,file = "chrX_transcripts_0.05.csv",row.names = F)
write.csv(results_genes_0.05,file = "chrX_genes_0.05.csv",row.names = F)

# DESeq2 差异分析
library(DESeq2)
# 读取reads count 数据
# row.names = 1 以第一列作为行名
countsTable <- read.delim("DE_data.tsv",header = F,row.names = 1)
# 添加列名
sampleNames <- paste0("sample",1:12)
colnames(countsTable) <- sampleNames
# 创建一个分组信息对象（数据框）
countsDesign <- data.frame(row.names = colnames(countsTable),
     group = c("FY","FG","FY","FG","FG","FY","MY","MY","MY","MG","MG","MG"))
head(countsDesign)
# 将reads count 数据由数据框转为矩阵类型
countsMatrix <- as.matrix(countsTable)
# 将矩阵转为DESeq2的数据类型dds
# 使用group来进行比较
dds <- DESeqDataSetFromMatrix(countsMatrix,colData = countsDesign,design = ~group)
# 除去所有值都为0的行
dds <- dds[rowSums(counts(dds))>0,]
# DESeq2 分析：大小因子估计、离差的估计、负二项分布的拟合以及计算相应的统计量
# parallel = T：进行并行计算，提高运行速度
dds <- DESeq(dds,parallel = T)
# 提取性别间差异表达基因的分析结果
sex_Y_res <- results(dds,contrast = c("group","FY","MY"),parallel = T)
sex_G_res <- results(dds,contrast = c("group","FG","MG"),parallel = T)
# 在群体间差异表达基因分析的结果
population_F_res <- results(dds,contrast = c("group","FY","FG"),parallel = T)
population_M_res <- results(dds,contrast = c("group","MY","MG"),parallel = T)
# 将性别间显著差表达的基因
Sig_sex_Y_0.01 <- sex_Y_res[which(sex_Y_res$padj < 0.01),]
write.csv(Sig_sex_Y_0.01,file = "Sig_sex_Y_0.01_DESeq2.csv")
# 提取种群间差异表达的基因
Sig_pop_F_0.01 <- population_F_res[which(population_F_res$padj < 0.01),]
write.csv(Sig_pop_F_0.01,file = "Sig_pop_F_0.01_DESeq2.csv")

# edgeR 差异表达分析
library(limma)
library(edgeR)
# 读取reads count 数据
# row.names = 1 以第一列作为行名
rawdata <- read.delim("DE_data.tsv",header = F,row.names = 1)
# 添加列名
sampleNames <- paste0("sample",1:12)
colnames(rawdata) <- sampleNames
# 创建一个分组信息对象（数据框）
countsDesign <- data.frame(row.names = colnames(rawdata),
    group = c("FY","FG","FY","FG","FG","FY","MY","MY","MY","MG","MG","MG"))
# 构建DGEList对象（edgeR所需的数据格式）
data <- DGEList(counts = rawdata,group = countsDesign$group)
# 过滤，至少有三个样本大于1
# cpm(data)：每一个reads数除以reads总数*1百万
keep <- rowSums(cpm(data)>1)>=3
# keep.lib.sizes 参数来控制归一化过程中是否保留样本大小因子
data <- data[keep,,keep.lib.sizes = F]
# 数据的标准化
data <- calcNormFactors(data)
# 创建设计矩阵
design <- model.matrix(~0+group,data = countsDesign)
head(design)
# 设计矩阵的列名
colnames(design) <- levels(factor(countsDesign$group))
# 估计方差
data <- estimateDisp(data,design)
# 构建DGEGLM对象
fit <- glmQLFit(data,design)
# 创建对照矩阵
my.contrasts <- makeContrasts(YRI.F.vs.M = FY - MY,GBR.F.vs.M = FG - MG,
                              F.YRI.vs.GBR = FY - FG,M.YRI.vs.GBR = MY - MG,levels = design)
my.contrasts
# 差异分析 QL F-test
qlf_YRI.F.vs.M <- glmQLFTest(fit,contrast = my.contrasts[,"YRI.F.vs.M"])
qlf_GBR.F.vs.M <- glmQLFTest(fit,contrast = my.contrasts[,"GBR.F.vs.M"])
qlf_F.YRI.vs.GBR <- glmQLFTest(fit,contrast = my.contrasts[,"F.YRI.vs.GBR"])
qlf_M.YRI.vs.GBR <- glmQLFTest(fit,contrast = my.contrasts[,"M.YRI.vs.GBR"])
# 对每个差异分析的p值进行校正（以qlf_YRI.F.vs.M为例）
YRI.F.vs.M_padj <- p.adjust(qlf_YRI.F.vs.M$table$PValue,method = "BH")
# 提取差异表达基因
Sig_YRI.F.vs.M <- qlf_YRI.F.vs.M$table[which(YRI.F.vs.M_padj < 0.05),]
# 保存到文件
write.csv(Sig_YRI.F.vs.M,"Sig_YRI.F.vs.M_0.05_edgeR.csv")
