生信实例

  • 将fastq文件转为fasta的实例
x
# NR表示行号，NR%4==2表示取出第2、6、10...的行，即序列的行
# print 表示的是输出的内容，">E"为普通字符，NR/4+0.5为数学运算，"\n"表示输出换行符，$0表示的数读入的序列
awk 'NR%4==2 {print ">E"NR/4+0.5"\n"$0;}' example.fq > example.fa
  • 查看fasta文件中序列的长度
x
# -v 表示的是取反，不输出'>'开头的行
# $0 表示读入的内容（有多列的话，第1列为$1，第2列为$2，以此类推）
grep -v '>' example.fa | awk '{print length($0);}'
  • 统计长度的分布
x
grep -v '>' example.fa | awk '{print length($0);}' | sort | uniq -c

sed

x
# 在每一行前面一行添加行号（如原来3列会变为6列）
cat <<END | sed =
# 为每一行的前一行添加行号
# 之后将下一行也一起读取进来（每次读两行，即一个N表示下面的一行），在最前面添加上 >A 
# 其中 s/ / /表示的是替换，即将 ^ 替换为 >A
sed = file | sed "N;s/^/>A/"
# 如：s/\n/\t/将换行符替换为制表符
x
# 每次读取三行，然后将前两行使用\t合并为一行
# 此时第2、3行的\n不会被替换
sed = file | sed "N;N;s/^/>A/"
# 全局替换
# 每次读取三行，然后将3行都是以\t合并为一行
sed = file | sed "N;N;s/^/>A/g"
x
# 现有一个多行的fasta文件（每个序列的序列不止一行），要将每个序列的多行序列变为一行，即ID+序列（两行，原来的话ID一行，序列可以有多行）
# tr：将换行符\n替换为制表符\t（此时整个文件变为一行）
# 第1个sed：将 > 全部替换为 \n> （此时每个序列都变为一行，同时第一行为空行）
# 第2个sed：将第一个制表符\t（ID和序列）替换为\n（此时每个序列为两行，即ID和序列分开）
# 第3个sed：将原来多行的序列拼接为一行
# 最后的grep的作用是把开头的一个空行删除
cat file.fa | tr "\n" "\t" | sed "s/>/\n>/g" | sed "s/\t/\n/" | sed "s/\t//g" | grep -v "^$" > new_file.fa
# 多个sed命令也可以使用-e连接
x
# 将文件的第2列作为数字进行排序，文件含有表头
# 从文件的第2行到结尾
tail -n +2 file.txt | sort -k2,2n | sed "1i 添加表头到最开始"
x
# 只输出偶数行
# n表示处理下一行而跳过当前行，p打印出来
echo -e "odd\neven\nodd\neven" | sed -n "n;p"
# 只输出奇数行
# 即将下一行读进来之后替换为空
echo -e "odd\neven\nodd\neven" | sed -n "N;s/\n.*//;p"

awk

  • 一行一行读取
  • 将以 \t 分割的文件的第一列和第二列调换位置。
x
# BIGIN表示初始化，只会执行一次，OFS指定列的分割符为\t
# 输出第2列和第1列
awk 'BIGIN{OFS="\t"}{print $2,$1}' file.txt
x
# 实现的效果是输出在pattern.txt里指出的ID的序列
# file.fa：为fasta文件，ID为一行，序列为一行
# pattern.txt：要查找的ID文件，一行一个ID
# ARGIND==1表示的是文件一（file.fa），ARGIND==2表示文件二（pattern.txt）
# $0~/>/：该行是否以 > 开头，name=$0：将该行赋值给变量name
# sub('>','',name)：将ID行的开头 > 删除，在原变量上修改（name）
# a[name]=$0：创建新的数组，将该行赋值给该数组
awk 'ARGIND==1{if($0~/>/) {name=$0; sub(">","",name); a[name]=$0;} else {a[name]=a[name]"\n"$0;}}ARGIND==2{print a[$0];}' file.fa pattern.txt
x
# 实现的结果是输出pattern.txt里列出的基因的ensemble ID
# maps.txt：有两列，第一列可以看作是基因symbol，第二列是ensemble ID
# pattern.txt：只有一列，为基因symbol
# BEGIN{OFS=FS="\t"}：初始化，输入和输出的列风格符否是制表符\t
awk 'BEGIN{OFS=FS="\t"}ARGIND==1{a[$1]=$2;}ARGIND==2{print a[$1];}' maps.txt pattern.txt
x
# 统计第二列中每个单词（每一行均为一个单词）出现的次数
# FNR > 1：从第二行开始（因为第一行为表头）
# END：在文件读取完后执行
awk 'BEGIN{OFS=FS="\t"}{if(FNR > 1){a[$2]+=1;}}END{for(i in a) print i,a[i];}' file.txt
# 类似效果
tail -n +2 file.txt | cut -f2 | sort | uniq -c
x
# 只保留2为小数
# file.txt：第一列为id(纯数字)，第一行为表头
awk 'BEGIN{OFS=FS="\t"}{if(FNR==1) print $0; else {$2=$2/3; printf "%d\t%.2f\n", $1,$2;}}' file.txt
x
# 有一个矩阵数据，既有行名，也有列名，要保留一位小数
# FS表示的是原文件的列分隔符（不一定是制表符），NF表示的是最后一行，RS表示的是原文件的换行符
awk '{if(FNR==1) {print $0;} else {printf "%s%s", $1,FS; for(i=2;i<=NF;i++){prinf "%.1f %s", $i, (i==NF?RS;FS)}}}' file.txt
x
# 判断fastq文件中的序列和质量值的字符数是否相等，输出不相等的
zcat file.fastq.gz | awk '{OFS=FS="\t"}{if(FNR%4==1) seq_name=$0; if(FNR%4==2) s_len=length($0); else if(FNR%4==0){q_len=length($0); if(s_len!=q_len) print seq_name;}}'
x
# 进行差异基因的筛选，第一列为ID，第二列为logFC，第三列为padj（含有表头）
awk 'BEGIN{OFS=FS="\t"}{if(FNR==1 || $3<=0.05) print $0;}' file.txt
# 也可以简化写法
awk 'FNR==1 || $3<=0.05' file.txt
# 可以分别指定上调或下调的输出文件
awk 'BEGIN{OFS=FS="\t";up="up.txt";dw="dw.txt";}{if(FNR==1){print $0 > up; print $0 > dw;}} .......'
# ARGV[1] 表示输入的我文件名
awk 'BEGIN{OFS=FS="\t";up=ARGV[1]"_up.txt";dw=ARGV[1]"_dw.txt";}{if(FNR==1){print $0 > up; print $0 > dw;}} .......'

GTF文件实战

  • gtf文件的开头注释以#号开头，接下来的每一行均为注释信息。
x
# 提取非注释行的第一条注释信息（非注释行的第一行），将该行变为一列，同时添加行号
grep -v "^#" file.gtf | head -n 1 | tr "\t" "\n" | awk '{OFS=FS="\t"}{print NR,$0;}'
x
# 为染色体编号前面添加chr（第一列）
grep -v "#" file.gtf | sed "s/^/chr/" > new_file.gtf
# 或
sed "/^[^#]/ s/^/chr/" file.gtf
x
# 统计染色体的种类数
# 因为gtf默认已经排序，因此不需要排序
# 得到的结果有两列 第一列为频数，第二列为染色体名字（同时单元格为右对齐，以空格分隔）
cut -f1 file.gtf | uniq -c > count.txt
# 将两列替换
# 值提取出拼装好的染色体
awk '/chr[0-9XYM]' count.txt | sed "s/  *\([0-9]*\) \(chr.*\)/\2\t\2"
x
# 统计基因的数目（文件的第3列为注释信息的类型，如gene、exon、transcript、CDS等）
cut -f file.gtf | sort | uniq -c
awk '{a[$3]+=1}END{for(i in a) print i,a[i];}'
x
# 计算GTF中外显子的总长度
# 使用第三列判断是否为外显子，第四列（起始）和第五列（结束）计算长度，再累加。
awk '{if($3=="exon") sum+=$5-$4+1;}END{print "总长度：",sum;}'
